display "Start";

i = 0;

while (i < 3) {

    j = 0;

    while (j < 3) {

        k = 0;

        while (k < 3) {

            x = i + j + k;

            if (x % 2 == 0) {
                display "even";
            }
            else {
                display "odd";
            }

            display x;

            k = k + 1;
        }

        j = j + 1;
    }

    i = i + 1;
}

display "End";
